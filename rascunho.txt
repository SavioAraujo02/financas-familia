### **💳 LÓGICA DE FATURA vs PARCELAS:**

```javascript
// CENÁRIO 1: Marcar FATURA como paga
if (marcarFaturaPaga) {
  // Busca TODAS as parcelas dessa fatura
  const parcelasNaFatura = getParcelas({
    cartao: 'Nubank',
    fatura: 'Jul/2025'
  });
  
  // Marca TODAS automaticamente
  parcelasNaFatura.forEach(parcela => {
    parcela.status = 'PAGA';
    parcela.dataPagamento = dataFatura;
  });
}

// CENÁRIO 2: Marcar PARCELA individual como paga
if (marcarParcelaPaga) {
  // Verifica se TODAS as parcelas da fatura foram pagas
  const todasParcelas = getParcelasFatura(parcela.fatura);
  const parcelasPagas = todasParcelas.filter(p => p.status === 'PAGA');
  
  if (parcelasPagas.length === todasParcelas.length) {
    // Marca a FATURA como paga automaticamente
    fatura.status = 'PAGA';
  }
}
```

## **⚙️ SISTEMA DE SINCRONIZAÇÃO:**

### **🔄 Auto-Update em Tempo Real:**

```javascript
// WEBHOOK de sincronização
const sincronizarFaturasParcelas = {
  
  // Quando marca fatura
  onFaturaPaga: (faturaId) => {
    const parcelas = database.parcelas.where({faturaId});
    parcelas.update({
      status: 'PAGA',
      dataPagamento: now(),
      sincronizadoPor: 'FATURA'
    });
    
    // Notifica usuários
    notificar(`💳 Fatura paga! ${parcelas.length} parcelas atualizadas`);
  },
  
  // Quando marca parcela
  onParcelaPaga: (parcelaId) => {
    const parcela = database.parcelas.find(parcelaId);
    const fatura = database.faturas.find(parcela.faturaId);
    
    // Verifica se todas parcelas foram pagas
    const totalParcelas = database.parcelas.where({faturaId: fatura.id});
    const parcelasPagas = totalParcelas.where({status: 'PAGA'});
    
    if (totalParcelas.length === parcelasPagas.length) {
      fatura.update({
        status: 'PAGA',
        dataPagamento: now(),
        sincronizadoPor: 'PARCELAS'
      });
      
      notificar(`🎉 Fatura ${fatura.nome} completamente paga!`);
    }
  }
};
```
